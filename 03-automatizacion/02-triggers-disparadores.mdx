<Callout kind="info">Los triggers definen qué evento inicia un flujo.</Callout>
<Tabs>
  <Tab label="Mensaje entrante">Se activa con cualquier mensaje del cliente.</Tab>
  <Tab label="Palabra clave">Solo con palabras específicas: "HORARIO", "AYUDA".</Tab>
  <Tab label="Botón/lista">Al presionar botón o seleccionar opción.</Tab>
  <Tab label="Programado">Fecha/hora específica o recurrente.</Tab>
  <Tab label="Webhook">Datos desde sistema externo vía POST.</Tab>
</Tabs>
